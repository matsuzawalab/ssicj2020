1 データベースに接続する

本章では，データベースに接続する方法を学びます．

1.1 データベースに接続し，select文を発行し，結果を出力する

<!DOCTYPE html>

<html>

<head>
    <meta http-equiv="Content-Type" content="text/html; charset=UTF-8">

    <script type="text/javascript" src="https://ajax.googleapis.com/ajax/libs/jquery/1.9.1/jquery.min.js"></script>

    <script>
        async function button1Pressed() {
            var userid = document.getElementById('tf1').value;
            var password = document.getElementById('tf2').value;

            var sql = 'select * from Tweets order by time desc limit 20';
            var query = {
                db: 'crecodb',
                userid: userid,
                password: password,
                sql: sql
            };

            var objects = await connect(query);
            console.log(objects);
            var html = '';
            html = html + '<ul>';
            for (var i = 0; i < objects.length; i++) {
                var object = objects[i];
                html = html + '<li>' + object.tweet + '</li>';
            }
            html = html + '</ul>';
            document.getElementById('result').innerHTML = html;
        }

        //この関数は変えないこと
        function connect(query) {
            return new Promise(function (resolve) {
                $.post('http://dbkiso.si.aoyama.ac.jp/api/', query, function (data, textStatus) {
                    var objects = JSON.parse(data);
                    resolve(objects);
                });
            });
        }
    </script>

</head>

2 データベースへのログイン機能を作る

本章では，ログイン機能とページ遷移のプログラムを学びます．

2.1 localStorageに情報を保存する

localStorageを使って情報をbrowserに保存する方法を学びます． 
<!DOCTYPE HTML>
<html>

<head>
    <meta charset="utf-8">
    <script type="text/javascript" src="https://ajax.googleapis.com/ajax/libs/jquery/1.9.1/jquery.min.js"></script>
    <script>
        function button1Pressed() {
            var userid = document.getElementById('tf1').value;
            var password = document.getElementById('tf2').value;

            window.localStorage.setItem('userid', userid);
            window.localStorage.setItem('password', password);
        }

        function button2Pressed() {
            var userid = window.localStorage.getItem('userid');
            var password = window.localStorage.getItem('password');

            document.getElementById('tf1').value = userid;
            document.getElementById('tf2').value = password;
        }
    </script>
</head>

<body>
    <input id="tf1" type="textfield">ログイン名
    <br>
    <input id="tf2" type="password">パスワード
    <button onclick="button1Pressed()">A</button>
    <button onclick="button2Pressed()">B</button>
    <hr>
    <p id="result"></p>
</body>

</html>
